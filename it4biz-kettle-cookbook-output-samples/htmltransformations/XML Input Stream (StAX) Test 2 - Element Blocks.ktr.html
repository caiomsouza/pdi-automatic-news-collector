This example parses the XML Input Stream (StAX) Test 2 - Element Blocks.xml
and has 2 main sample data blocks (Analyzer Lists & Products).

The different data blocks are separated by splitting the parent XML path to levels and the Switch / Case steps.
This separation can also be solved by the 'string contains' option of the Switch / Case step or other steps.

In more complex processings, Mappings (sub transformations) should be used for the different data blocks
to have it clearly represented.

private int pos_xml_element_level;

private Long xml_data_type_numeric;
private Long xml_element_level;
private String xml_data_name;
private String xml_path;

private final static String DEFAULT="default";
private String set_value=DEFAULT;

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException 
{
  Object[] r=getRow();
  if (r==null)
  {
    setOutputDone();
	return false;
  }

  if (first) {
	 // get field positions
     pos_xml_data_type_numeric = getInputRowMeta().indexOfValue("xml_data_type_numeric");
     pos_xml_element_level = getInputRowMeta().indexOfValue("xml_element_level");
     pos_xml_data_name = getInputRowMeta().indexOfValue("xml_data_name");
	 pos_xml_data_value = getInputRowMeta().indexOfValue("xml_data_value");
     pos_xml_path = getInputRowMeta().indexOfValue("xml_path");
     if (pos_xml_data_type_numeric<0 || pos_xml_element_level<0 || pos_xml_data_name<0 || pos_xml_data_value<0 || pos_xml_path<0) {
         throw new KettleException("One of the xml_ fields was not found.");
     }
	 
     // get parameters
	 xml_data_type_numeric=new Long(getParameter("xml_data_type_numeric"));
     xml_element_level=new Long(getParameter("xml_element_level"));
     xml_data_name=getParameter("xml_data_name");
     xml_path=getParameter("xml_path");

     first=false;
  }
 
  // set
  if(r[pos_xml_data_type_numeric].equals(xml_data_type_numeric)) {
    if(r[pos_xml_element_level].equals(xml_element_level)) {
      if(r[pos_xml_data_name].equals(xml_data_name)) {
        if(r[pos_xml_path].equals(xml_path)) {
          set_value=(String)r[pos_xml_data_value];
        }
      }
    }
  }
  // reset when level goes down again
  if(((Long)r[pos_xml_element_level]).compareTo(xml_element_level)<0) set_value=DEFAULT;

  // find it in the Target steps definitions
  RowSet targetRowSet=null;
  try {
    targetRowSet=findTargetRowSet(set_value);
  } catch (Exception e) {
    // not found: set it to default
    targetRowSet=findTargetRowSet(DEFAULT);
  }

  putRowTo(getInputRowMeta(), r, targetRowSet);

  return true;
}

// Repeat id and name in each row

private int pos_xml_data_name;
private int pos_xml_data_value;

private int pos_new_field_1;
private int pos_new_field_2;

private String xml_data_name_1;
private String xml_data_name_2;

private String last_value_1;
private String last_value_2;

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException 
{
  Object[] r=getRow();
  if (r==null)
  {
    setOutputDone();
	return false;
  }

  if (first) {
	 // get field positions (input)
     pos_xml_data_name = getInputRowMeta().indexOfValue("xml_data_name");
	 pos_xml_data_value = getInputRowMeta().indexOfValue("xml_data_value");

     // get parameters
     xml_data_name_1=getParameter("xml_data_name_1");
     xml_data_name_2=getParameter("xml_data_name_2");

	 // get field positions (output)
     pos_new_field_1 = data.outputRowMeta.indexOfValue(xml_data_name_1);
     pos_new_field_2 = data.outputRowMeta.indexOfValue(xml_data_name_2);

     first=false;
  }
 
  // set
  if(r[pos_xml_data_name].equals(xml_data_name_1)) {
    last_value_1=(String)r[pos_xml_data_value];
  }
  if(r[pos_xml_data_name].equals(xml_data_name_2)) {
    last_value_2=(String)r[pos_xml_data_value];
  }

  Object[] outputRowData = createOutputRow(r, data.outputRowMeta.size());

  // append it to all rows
  outputRowData[pos_new_field_1]=new String(Const.NVL((String)last_value_1,""));
  outputRowData[pos_new_field_2]=new String(Const.NVL((String)last_value_2,""));

  putRow(data.outputRowMeta, outputRowData);

  return true;
}
