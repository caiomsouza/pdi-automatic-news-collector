private int pos_xml_data_type_numeric;
private int pos_xml_data_name;
private int pos_xml_data_value;
private int pos_xml_path;
private int pos_ProductParentID;
private int pos_ProductID;
private int pos_ProductLevel;

private int productLevel=0;
private String productID[]=new String[1000];

private final static Long START_ELEMENT=new Long(1);
private final static Long END_ELEMENT=new Long(2);
private final static Long ATTRIBUTE=new Long(10);

private final static String ELEMENT_PRODUCT="Product";
private final static String ATTR_ID="id";

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException 
{
  Object[] r=getRow();

  if (r==null)
  {
    setOutputDone();
	return false;
  }

  if (first) {
	 // get field positions (input)
     pos_xml_data_type_numeric = getInputRowMeta().indexOfValue("xml_data_type_numeric");
     pos_xml_data_name = getInputRowMeta().indexOfValue("xml_data_name");
	 pos_xml_data_value = getInputRowMeta().indexOfValue("xml_data_value");
     pos_xml_path = getInputRowMeta().indexOfValue("xml_path");

	 // get field positions (output)
     pos_ProductID = data.outputRowMeta.indexOfValue("ProductID");
     pos_ProductParentID = data.outputRowMeta.indexOfValue("ProductParentID");
     pos_ProductLevel = data.outputRowMeta.indexOfValue("ProductLevel");

     if (pos_xml_data_type_numeric<0 || pos_xml_data_name<0 || pos_xml_data_value<0 || pos_xml_path<0 || pos_ProductID<0 || pos_ProductParentID<0 || pos_ProductLevel<0 ) {
         throw new KettleException("One of the xml_ fields or output fields was not found.");
     }
	 
     first=false;
  }

  boolean ignoreLine=false;

   // START_ELEMENT: increment productLevel
  if(r[pos_xml_data_type_numeric].equals(START_ELEMENT)) {
      ignoreLine=true;
      if(r[pos_xml_data_name].equals(ELEMENT_PRODUCT)) {
		productLevel++;
      }
  }

 // END_ELEMENT: decrement productLevel
  if(r[pos_xml_data_type_numeric].equals(END_ELEMENT)) {
      ignoreLine=true;
      if(r[pos_xml_data_name].equals(ELEMENT_PRODUCT)) {
		productLevel--;
      }
  }

  // productID
  if(r[pos_xml_data_type_numeric].equals(ATTRIBUTE)) {
      if(r[pos_xml_data_name].equals(ATTR_ID)) {
		productID[productLevel]=new String((String)r[pos_xml_data_value]);
        ignoreLine=true;
      }
  }

  if(!ignoreLine) {
    Object[] outputRowData = createOutputRow(r, data.outputRowMeta.size());
    outputRowData[pos_ProductID]=new String(Const.NVL((String)productID[productLevel],""));
    if(productLevel>0) {
      outputRowData[pos_ProductParentID]=new String(Const.NVL((String)productID[productLevel-1],""));
    } else {
       outputRowData[pos_ProductParentID]=null;
    }
    outputRowData[pos_ProductLevel]=new Long(productLevel);

    putRow(data.outputRowMeta, outputRowData);
  }

  return true;
}
