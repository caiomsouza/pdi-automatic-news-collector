var meta = row[getInputRowMeta().indexOfValue("meta")];
var object_name = meta.getName();
var object_description = meta.getDescription();
var object_extended_description = meta.getExtendedDescription();
var used_databases = "";

if (FileType.equalsIgnoreCase("Transformation")) {
	var used_databases_first=true;
	for (i=0;i<meta.nrDatabases();i++) {
  		var db = meta.getDatabase(i);
  		if (meta.isDatabaseConnectionUsed(db)) {
    		if (!used_databases_first) {
				used_databases+=", ";
			} else {
				used_databases_first=false;
		    }
    		used_databases+=db.getName();
		  }
	}
} else {
	for (i=0;i<meta.nrDatabases();i++) {
  		var db = meta.getDatabase(i);
    	if (i>0) {
			used_databases+=", ";
		}
		used_databases+=db.getName();
	}
}
