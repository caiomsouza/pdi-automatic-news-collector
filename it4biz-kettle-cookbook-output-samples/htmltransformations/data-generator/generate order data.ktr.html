Script 1
// The ordernr is linked to the customerId and the orderSeq.
// it starts at 5000000 and is then incremented each time
// there is a new customerId/orderSeq combination

var prevId;
var prevSeq;
var orderNr

if (prevId == null)
{
  orderNr = 5000000;
}
else
{
  if (prevId != customerId.getInteger() || prevSeq != orderSeq.getInteger())
  {
    orderNr++;
  }
}

prevId = customerId.getInteger();
prevSeq = orderSeq.getInteger();
